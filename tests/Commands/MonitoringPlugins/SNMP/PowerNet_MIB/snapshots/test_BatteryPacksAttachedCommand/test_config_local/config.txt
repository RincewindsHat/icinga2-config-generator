object CheckCommand "command_instance_local" {
  command = [ "$monitoring_plugin_dir$" + "/SNMP/PowerNet_MIB/check_battery_packs_attached.py"]
  arguments = {
    "-u" = {
      value = "$command_snmp_username$"
      required = true
    }
    "-p" = {
      value = "$command_snmp_password$"
      required = true
    }
    "-H" = {
      value = "$command_snmp_host$"
      required = true
    }
    "--timeout" = {
      value = "$command_snmp_timeout$"
      set_if = {{ macro("$command_snmp_timeout$") != false }}
    }

    "-c" = {
      value = "$command_battery_packs_attached_count$"
      required = true
    }

  }

}
