object CheckCommand "command_instance_local" {
  command = [ "$monitoring_plugin_dir$" + "/SNMP/Synology/check_temperature.py"]
  arguments = {
    "-u" = {
      value = "$command_snmp_username$"
      required = true
    }
    "-p" = {
      value = "$command_snmp_password$"
      required = true
    }
    "-H" = {
      value = "$command_snmp_host$"
      required = true
    }
    "--timeout" = {
      value = "$command_snmp_timeout$"
      set_if = {{ macro("$command_snmp_timeout$") != false }}
    }

    "-w" = {
      value = "$command_temperature_warning$"
      set_if = {{ macro("$command_temperature_warning$") != false }}
    }
    "-c" = {
      value = "$command_temperature_critical$"
      set_if = {{ macro("$command_temperature_critical$") != false }}
    }

  }

}
